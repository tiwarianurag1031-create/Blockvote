function showToast(type, msg) {
  const t = document.getElementById('toast');
  t.className = 'toast show ' + type;

const sa = a => a ? a.slice(0,6)+'...'+a.slice(-4) : '';
const fe = w => parseFloat(ethers.formatEther(w)).toFixed(4);
const pc = (v,t) => t===0 ? 0 : Math.round((v/t)*100);
function fmtTime(s) {
  if(s<=0) return 'Ended';
  const d=Math.floor(s/86400),h=Math.floor((s%86400)/3600),m=Math.floor((s%3600)/60);
  if(d>0) return d+'d '+h+'h';
  if(h>0) return h+'h '+m+'m';
  return m+'m '+(s%60)+'s';
}

async function connectWallet() {

    await provider.send("eth_requestAccounts",[]);

    if(net.chainId !== BigInt(parseInt(SID,16))) {

    signer = await provider.getSigner();

async function loadPolls() {
  if(!ct) return;
  document.getElementById('loadBox').style.display = 'block';
  document.getElementById('pollsGrid').innerHTML = '';
  document.getElementById('noPolls').style.display = 'none';
  try {
    const count = Number(await ct.getPollCount());
    polls = []; voted = {};
    for(let i=count-1; i>=Math.max(0,count-20); i--) {
      try {
        const [q,os,vc,et,fin,creator,tf,bo] = await ct.getPoll(i);
        const v = vc.map(x=>Number(x));
        const tv = v.reduce((a,b)=>a+b,0);
        const rem = Number(et) - Math.floor(Date.now()/1000);
        polls.push({id:i,question:q,options:Array.from(os),votes:v,endTime:Number(et),finalized:fin,creator,totalFunds:tf,betOption:Number(bo),totalVotes:tv,remaining:rem});
        if(walletAddr) { try { voted[i] = await ct.hasVoted(i,walletAddr); } catch{} }
      } catch{}
    }
    document.getElementById('sTotal').textContent = polls.length;
    document.getElementById('sActive').textContent = polls.filter(p=>!p.finalized&&p.remaining>0).length;
    document.getElementById('sDone').textContent = polls.filter(p=>p.finalized).length;
    document.getElementById('sVotes').textContent = polls.reduce((a,p)=>a+p.totalVotes,0);
    renderPolls();
  } catch(e) {
    showToast('error','Failed to load polls');

function pickOpt(i) {
  if(!curPoll) return;
  const rem = curPoll.endTime - Math.floor(Date.now()/1000);
  if(!(rem>0 && !curPoll.finalized) || voted[curPoll.id]) return;
  selOpt = i;
  curPoll.options.forEach((_,j) => {
    const r = document.getElementById('or'+j);
    if(r) r.classList.toggle('selected', j===i);
  });
  const vb = document.getElementById('voteBtn');
  if(vb) { vb.disabled=false; vb.textContent='🗳️ Vote (0.001 ETH) on "'+curPoll.options[i]+'"'; }
}

async function doVote() {
  if(selOpt===null || !ct) return;
  closeModal();
  showToast('info','Submitting vote...');
  try {
    const tx = await ct.vote(curPoll.id, selOpt, {value:ethers.parseEther("0.001")});
    showToast('info','Waiting for confirmation...');
    await tx.wait();
    showToast('success','Vote cast! ✅');
    loadPolls();
  } catch(e) { showToast('error', e.reason || e.message?.slice(0,80) || 'Vote failed'); }
}

async function doFinalize() {
  if(!ct) return;
  closeModal();
  showToast('info','Finalizing poll...');
  try {
    const tx = await ct.finalizePoll(curPoll.id);
    await tx.wait();
    showToast('success','Poll finalized! 🏆');
    loadPolls();
  } catch(e) { showToast('error', e.reason || e.message?.slice(0,80) || 'Finalize failed'); }
}

async function submitPoll() {

  if(validOpts.length<2){showToast('error','Need at least 2 options');return;}
  if(!ct){showToast('error','Connect wallet first');return;}
  const dur=parseInt(document.getElementById('pollDur').value);
  const bi=betOn?betIdx:255;
  const total=betOn?"0.006":"0.005";

  showToast('info','Confirm transaction in MetaMask...');
  try {
    const tx=await ct.createPoll(q,validOpts,dur,bi,{value:ethers.parseEther(total)});

    showPage('polls');
  } catch(e){showToast('error',e.reason||e.message?.slice(0,80)||'Failed');}
  btn.disabled=false; btn.textContent='🚀 Create Poll on Blockchain';
}
